[RelayCommand]

    public void RemoveContactFromList(IPerson person)
    {
        if (person != null)
        {
            bool result = _personRepository.RemovePersonFromList(person.Email);
            if (result)
            {
                UpdateCustomerList();
            }

        }

    }
